class AutoWhiteBalancing
{
    public:
    AutoWhiteBalancing()
        : m_previousCCT(0.0)
        , m_previousTint(0.0)
        , m_maxIteration(30)
        , m_stopTolerance(10.0)
    {
        
    }
    
    virtual ~AutoWhiteBalancing()
    {
        
    }
    
    pixie::core::xyY FindNeutralGrayIteratively(const cv::Mat& bayerImage, const pixie::core::MetadataC& metadata, pixie::core::xyY xyNeutralGray, double dampingParameter = 0.8, const cv::Mat& darkAreaMask = cv::Mat())
    {
        bool bStopIteration = false;
        
        for(size_t i = 0; i < m_maxIteration; i++)
        {
            cv::Mat linearSrgbImage = renderStage1ImageToSrgbLinear(bayerImage, metadata, false, xyNeutralGray);
            
            cv::imshow(linearSrgbImage, 200);
            
            cv::Matx31d xyzNeutralGray = srgbLinearToXyzAverage(linearSrgbImage,xyNeutralGray,darkAreaMask);
            xyNeutralGray = XyzToXyY(xyzNeutralGray);
            
            pixie::core::Algorithms::ColorTemperature ct;
            double cct = ct.ForXy(xyNeutralGray).GetTemperature();
            double tint = ct.ForXy(xyNeutralGray).GetTint();

            if(std::abs(cct - m_previousCCT) < m_stopTolerance && std::abs(tint - m_previousTint) < m_stopTolerance)
            {
                bStopIteration = true;
            }
            
            m_previousCCT = m_previousCCT + dampingParameter*(cct - m_previousCCT);
            m_previousTint = m_previousTint + dampingParameter*(tint - m_previousTint);
            
            std::cout << "Temperature: " << m_previousCCT << " K, Tint: " << m_previousTint << std::endl;
            
            xyNeutralGray = ct.ForTemperatureAndTint(m_previousCCT, m_previousTint).GetXy();
            
            if (bStopIteration == true)
            {
                break;
            }
        }
        
        return xyNeutralGray;
    }
    
    private:
    
    double m_previousCCT;
    double m_previousTint;
    
    unsigned int m_maxIteration;
    double m_stopTolerance;
    
    cv::Mat renderStage1ImageToSrgbLinear(const cv::Mat& bayerImage, pixie::core::MetadataC metadata, bool bIgnoreBorders, pixie::core::xyY targetWP)
    {
        pixie::core::ColorSpaceT outputColorspace = pixie::core::CS_SRGB_LINEAR;
        bool useExtendedBayering = true;
               
        if(bIgnoreBorders == true)
        {
            metadata.m_activeArea = cv::Rect(0,0,4000,3000);
            metadata.m_cropArea = cv::Rect(0,0,4000,3000);
        }
        
        pixie::IPIPE::OfficeWorkflowRenderingPipeline officePipeline(
            metadata, 
            targetWP, 
            useExtendedBayering, 
            outputColorspace);
        cv::Mat bayerImage_temp;
        bayerImage.copyTo(bayerImage_temp);
        cv::Mat rgbImage = officePipeline.Apply(bayerImage_temp);
        return rgbImage;
    }
    
    cv::Matx31d srgbLinearToXyzAverage(const cv::Mat& srgbLinear,pixie::core::xyY targetWpInXyY, const cv::Mat& darkAreaMask = cv::Mat())
    {
        auto meanValues = cv::mean(srgbLinear, darkAreaMask);
        auto grayValuesInSrgbLinear = cv::Matx31d(meanValues(0),meanValues(1),meanValues(2));
        
        pixie::core::ColorSpaceT outputColorSpace = pixie::core::CS_SRGB_LINEAR;
        
        pixie::IPIPE::Blocks::NestedColorTransformation nst;
        
        nst
            .Create<pixie::IPIPE::Blocks::ChromaticAdaption>()
                .From(targetWpInXyY)
                .To(pixie::core::WB_D65)
            .Add()
            .Create<pixie::IPIPE::Blocks::FinalColorSpaceTransformation>()
                    .FromXyzD65ColorSpace()
                    .WithFinalColorSpace(outputColorSpace)
            .Add();
        auto sRgbLinearToXyz = nst.GetInverseColorTransformationMatrix();
        auto xyzNeutralGray = sRgbLinearToXyz * grayValuesInSrgbLinear;

        auto xyzNeutralGrayScaled = cv::Matx31d(xyzNeutralGray(0)/xyzNeutralGray(1),xyzNeutralGray(1)/xyzNeutralGray(1),xyzNeutralGray(2)/xyzNeutralGray(1));
        return xyzNeutralGrayScaled;
    }
    
    cv::Matx31d XyYToXyz(const pixie::core::xyY& Xy, double Y)
    {
        double iy = 1 / (Xy.y + 1e-6);

        double X = Xy.x * Y * iy;
        double Z = Y * (1-Xy.x-Xy.y) * iy;

        cv::Matx31d xyz = cv::Matx31d(X,Y,Z);
    
        return xyz;
    }
    
    pixie::core::xyY XyzToXyY(const cv::Matx31d& Xyz)
    {
        double denominator = Xyz(0) + Xyz(1) + Xyz(2);
        pixie::core::xyY xyY;
        xyY.x = Xyz(0) / denominator;
        xyY.y = Xyz(1) / denominator;
        return xyY;
    }
};
